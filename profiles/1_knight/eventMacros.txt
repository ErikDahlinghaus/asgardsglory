automacro attackGoodMinibosses {
    MobNear /(Angeling|Deviling)/
    timeout 1
    call {
        do a $.MobNearLastBinId
        stop
    }
}

automacro mimic {
    MobNearDist /Mimic/ < 7
    disabled 1
    call {
        do a $.MobNearDistLastBinId
        stop
    }
}

automacro dropMagnifier {
    InInventory "Magnifier" > 0
    CharCurrentWeight > 47%
    InCity 0
    timeout 1
    call {
        [
            $id = &inventory ($.InInventoryLast)
            do drop $id
        ]
        stop
    }
}
